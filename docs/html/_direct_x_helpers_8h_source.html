215 constexpr bool IsPowerOf2(T x) noexcept { return ((x != 0) && !(x & (x - 1))); }

217 // Helpers for aligning values by a power of 2
218 template<typename T>
219 inline T AlignDown(T size, size_t alignment) noexcept
220 {
221 if (alignment > 0)
222 {
223 assert(((alignment - 1) & alignment) == 0);
224 auto mask = static_cast<T>(alignment - 1);
225 return size & ~mask;
226 }
227 return size;
228 }
229
230 template<typename T>
231 inline T AlignUp(T size, size_t alignment) noexcept
232 {
233 if (alignment > 0)
234 {
235 assert(((alignment - 1) & alignment) == 0);
236 auto mask = static_cast<T>(alignment - 1);
237 return (size + mask) & ~mask;
238 }
239 return size;
240 }
241 }
