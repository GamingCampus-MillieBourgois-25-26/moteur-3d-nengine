70 inline void Quaternion<T>::SetFromToRotation(const Vector3<T>& from, const Vector3<T>& to)
71 {
72 Vector3<T> f = from.normalized();
73 Vector3<T> t = to.normalized();
74
75 T dot = Vector3<T>::Dot(f, t);
76 if (dot >= 1.0f) {
77 Set(1, 0, 0, 0);
78 return;
79 }
80 if (dot <= -1.0f) {
81 Vector3<T> ortho = Vector3<T>::OrthoNormalize(f, Vector3<T>(1, 0, 0));
82 if (ortho.sqrMagnitude() < 1e-6) {
83 ortho = Vector3<T>::OrthoNormalize(f, Vector3<T>(0, 1, 0));
84 }
85 Set(0, ortho.x, ortho.y, ortho.z);
86 return;
87 }
88
89 Vector3<T> cross = Vector3<T>::Cross(f, t);
90 T s = std::sqrt((1 + dot) * 2);
91 T invs = 1 / s;
92
93 Set(s * 0.5f, cross.x * invs * 0.5f, cross.y * invs * 0.5f, cross.z * invs * 0.5f);
94 Normalize();
95 }

97 template<typename T>
98 inline void Quaternion<T>::SetLookRotation(const Vector3<T>& forward, const Vector3<T>& up)
99 {
100 Vector3<T> f = forward.normalized();
101 Vector3<T> u = up.normalized();
102 Vector3<T> r = Vector3<T>::Cross(u, f).normalized();
103 u = Vector3<T>::Cross(f, r);
104
105 T m00 = r.x, m01 = r.y, m02 = r.z;
106 T m10 = u.x, m11 = u.y, m12 = u.z;
107 T m20 = -f.x, m21 = -f.y, m22 = -f.z;
108
109 T trace = m00 + m11 + m22;
110 if (trace > 0) {
111 T s = 0.5f / std::sqrt(trace + 1.0f);
112 Set(
113 0.25f / s,
114 (m21 - m12) * s,
115 (m02 - m20) * s,
116 (m10 - m01) * s
117 );
118 }
119 else {
120 if (m00 > m11 && m00 > m22) {
121 T s = 2.0f * std::sqrt(1.0f + m00 - m11 - m22);
122 Set(
123 (m21 - m12) / s,
124 0.25f * s,
125 (m01 + m10) / s,
126 (m02 + m20) / s
127 );
128 }
129 else if (m11 > m22) {
130 T s = 2.0f * std::sqrt(1.0f + m11 - m00 - m22);
131 Set(
132 (m02 - m20) / s,
133 (m01 + m10) / s,
134 0.25f * s,
135 (m12 + m21) / s
136 );
137 }
138 else {
139 T s = 2.0f * std::sqrt(1.0f + m22 - m00 - m11);
140 Set(
141 (m10 - m01) / s,
142 (m02 + m20) / s,
143 (m12 + m21) / s,
144 0.25f * s
145 );
146 }
147 }
148 Normalize();

280 template<typename T>
281 Quaternion<T> Quaternion<T>::SlerpUnclamped(const Quaternion<T>& a, const Quaternion<T>& b, T t) {
282 Quaternion<T> qa = a.normalized();
283 Quaternion<T> qb = b.normalized();
284
285 T dot = Dot(qa, qb);
286
287 if (dot < 0.0f) {
288 qb = Quaternion(-qb.w, -qb.x, -qb.y, -qb.z);
289 dot = -dot;
290 }
291
292 if (dot > 0.9995) {
293 Quaternion<T> result = Quaternion(
294 qa.w + t * (qb.w - qa.w),
295 qa.x + t * (qb.x - qa.x),
296 qa.y + t * (qb.y - qa.y),
297 qa.z + t * (qb.z - qa.z)
298 );
299 result.Normalize();
300 return result;
301 }
302
303 T theta_0 = std::acos(dot);
304 T theta = theta_0 * t;
305 T sin_theta = std::sin(theta);
306 T sin_theta_0 = std::sin(theta_0);
307
308 Quaternion<T> qc = (qa * std::cos(theta) + (qb - qa * dot) * (sin_theta / sin_theta_0));
309 return qc;
310 }
