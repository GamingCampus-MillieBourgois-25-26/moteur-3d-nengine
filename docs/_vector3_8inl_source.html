169 template <typename T>
170 Vector3<T> Vector3<T>::Slerp(const Vector3& from, const Vector3& to, T t)
171 {
172 t = std::clamp(t, 0.f, 1.f);
173
174 Vector3 a = from.normalized();
175 Vector3 b = to.normalized();
176
177 T dot = Dot(a, b);
178 dot = std::clamp(dot, -1.0f, 1.0f);
179
180 T theta = std::acos(dot) * t;
181
182 Vector3 relative = (b - a * dot).normalized();
183
184 return a * std::cos(theta) + relative * std::sin(theta);
185 }

246 T Vector3<T>::Angle(const Vector3& v1, const Vector3& v2) {
247 T mag1 = v1.magnitude();
248 T mag2 = v2.magnitude();
249 if (mag1 == 0 || mag2 == 0) {
250 std::cerr << "Error: null vector has no angle defined\n";
251 return 0;
252 }
253 T dot = Dot(v1, v2) / (mag1 * mag2);
254 dot = std::clamp(dot, -1.0, 1.0);
255 return std::acos(dot);
